1-2 使用beautifulsoup4
from bs4 import BeautifulSoup # 导入模块
使用beautifulsoup来查看<h1>内容
bsObj = BeautifulSoup(html.read()) 
print(bsObj.h1)

会运行几秒,我们可以看到提示：

# <h1>An Interesting Title</h1>
#html内容就被取出来了

此外还有可以提取网站的其他内容

print(bsObj.div)
print(bsObj.body)

对于爬虫的代码设计时候，思考代码的总体格局，让代码既可以捕捉异常又容易阅读，这是很重要
的。如果你还希望能够很大程度地重用代码，那么拥有像getSiteHTML 和getTitle 这样的
通用函数（具有周密的异常处理功能）会让快速稳定地网络数据采集变得简单易行。

from urllib.request import urlopen
from urllib.error import HTTPError
from bs4 import BeautifulSoup


def getTitle(url):
    try:
        html = urlopen(url)
    except HTTPError as e:
        return None
    try:
        bsObj = BeautifulSoup(html.read())
        title = bsObj.body.h1
    except AttributeError as e:
        return None
    return title


title = getTitle("http://www.pythonscraping.com/pages/page1.html")
if title == None:
    print("Title could not be found")
else:
    print(title)
